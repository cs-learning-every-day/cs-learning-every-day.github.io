```c++
class Solution {
public:

    struct Cmp {
        bool operator() (ListNode* a, ListNode* b) {
            return a->val > b->val;
        }
    };

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        auto dummy = new ListNode();
        auto p = dummy;
        priority_queue<ListNode*, vector<ListNode*>, Cmp> min_heap;
        for (auto& item : lists) {
            if (item) {
                min_heap.push(item);
            }
        }
        while (min_heap.size()) {
            auto tmp = min_heap.top();
            min_heap.pop();
            p->next = tmp;
            if (tmp->next) min_heap.push(tmp->next);
            p = tmp;
        }
        return dummy->next;
    }
};
```

```java
class Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        ListNode dummy = new ListNode();
        PriorityQueue<ListNode> minHeap = new PriorityQueue<>((a, b) -> a.val - b.val);
        for (var l : lists) {
            if (l != null) { // 示例3 特例
                minHeap.offer(l);
            }
        }

        ListNode tail = dummy;
        while (!minHeap.isEmpty()) {
            ListNode cur = minHeap.poll();
            tail.next = cur;
            tail = tail.next;
            
            cur = cur.next;
            if (cur != null) {
                minHeap.offer(cur);
            }
        }
        return dummy.next;
    }
}
```
