    public ListNode reverseKGroup(ListNode head, int k) {
        ListNode dummy = new ListNode();
        dummy.next = head;
        for (ListNode p = dummy;;) {
            int n = 0;
            for (ListNode t = p.next; t != null; t = t.next) {
                n++;
            }
            if (n < k) {
                break;
            }
            // 翻转k个节点
            ListNode a = p.next, b = p.next.next;
            for (int i = 0; i < k - 1; i++) {
                var c = b.next;
                b.next = a;
                a = b;
                b = c;
            }
            var c = p.next;
            p.next = a;
            c.next = b;
            p = c;
        }
        return dummy.next;
    }
