
> 注意审题是left位置,right位置

思路：翻转[left,right]位置, 后续特殊处理

```c++
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        auto sentinel = new ListNode(-1, head);
        auto p = sentinel;
        for (int i = 0; i < left - 1; i++) {
            p = p->next;
        }

        auto front = p->next, back = front;
        for (int i = 0; i < right - left; i++) {
            back = back->next;
        }
        auto q = back->next;

        // reverse [front, back]
        auto a = front, b = a->next;
        for (int  i = 0; i < right - left; i++) {
            auto tmp = b->next;
            b->next = a;
            a = b; b = tmp;
        }

        p->next = back;
        front->next = q;
        return sentinel->next;
    }
```
