class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        ListNode* sentinel = new ListNode(-1);
        ListNode* p = sentinel;
        while (head) {
            if (head->next && head->next->val == head->val) {
                while (head->next && head->next->val == head->val) {
                    head = head->next;
                }
                p->next = head->next;
            } else {
                p->next = head;
                p = p->next;
            }
            head = head->next;
        }
        return sentinel->next;
    }
};
