    public ListNode swapPairs(ListNode head) {
        ListNode dummy = new ListNode();
        dummy.next = head;
        for (ListNode p = dummy; p.next != null && p.next.next != null;) {
            ListNode a = p.next;
            ListNode b = p.next.next;
            p.next = b;
            a.next = b.next;
            b.next = a;
            p = a;
        }
        return dummy.next;
    }
