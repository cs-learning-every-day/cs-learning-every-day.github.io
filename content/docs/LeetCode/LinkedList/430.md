/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* prev;
    Node* next;
    Node* child;
};
*/

class Solution {
public:
    Node* flatten(Node* head) {
        if (!head) return head;

        auto dummy = new Node();
        auto p = dummy;
        stack<Node*> st;
        st.push(head);
        while (st.size()) {
            auto tmp = st.top();
            st.pop();
            if (tmp->next) st.push(tmp->next);
            if (tmp->child) st.push(tmp->child);
            p->next = tmp;
            tmp->prev = p;
            tmp->child = nullptr;
            p = tmp;
        }

        dummy->next->prev = nullptr;
        return dummy->next;
    }
};
