    public ListNode rotateRight(ListNode head, int k) {
        if (head == null) return head;
        int n = 0;
        ListNode tail = head;
        for (var p = head; p != null; p = p.next) {
            n++;
            tail = p;
        }

        k = k % n;
        if (k == 0) return head;

        ListNode p = head;
        for (int i = 0; i < n - 1 - k; i++) {
            p = p.next;
        }

        tail.next = head;
        head = p.next;
        p.next = null;
        return head;
    }
