    public ListNode removeNthFromEnd(ListNode head, int k) {
        ListNode dummy = new ListNode();
        dummy.next = head;
        int n = 0;
        while (head != null) {
            n++;
            head = head.next;
        }

        ListNode prev = dummy;
        for (int i = 0; i < (n - k); i++) {
            prev = prev.next;
        }
        prev.next = prev.next.next;
        return dummy.next;
    }
