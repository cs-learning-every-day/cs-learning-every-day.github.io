
递归
```c++
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr || head->next == nullptr) return head;
        ListNode* tmp = reverseList(head->next);
        head->next->next = head;
        head->next = nullptr;
        return tmp;
    }
```

迭代
```c++
    ListNode* reverseList(ListNode* head) {
        ListNode* dummy = new ListNode();
        ListNode* cur = head;
        while (cur != nullptr) {
            auto tmp = cur->next;
            cur->next = dummy->next;
            dummy->next = cur;
            cur = tmp;
        }
        return dummy->next;
    }
```
