```c++
    ListNode* partition(ListNode* head, int x) {
        auto p1 = new ListNode();
        auto c1 = p1;
        auto p2 = new ListNode();
        auto c2 = p2;
        for (auto p = head; p; p = p->next) {
            if (p->val < x) {
                c1->next = p;
                c1 = p;
            } else {
                c2->next = p;
                c2 = p;
            }
        }
        c1->next = p2->next;
        c2->next = nullptr;
        return p1->next;
    }
```
