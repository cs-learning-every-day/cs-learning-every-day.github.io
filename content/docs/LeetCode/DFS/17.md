使用StringBuilder优化字符串copy

```java
class Solution {
    List<String> res = new ArrayList<>();
    String[] keyboards = new String[]{
        "", "",
        "abc", "def",
        "ghi", "jkl", "mno",
        "pqrs", "tuv", "wxyz",
    };

    public List<String> letterCombinations(String digits) {
        if (digits == null || digits.length() == 0) {
            return res;
        }
        dfs(digits, 0, digits.length(), new StringBuilder());
        return res;
    }

    private void dfs(String digits, int pos, int len, StringBuilder cur) {
        if (pos == len) {
            res.add(cur.toString());
            return;
        } else {
            for (char c : keyboards[digits.charAt(pos) - '0'].toCharArray()) {
                cur.append(c);
                dfs(digits, pos + 1, len, cur);
                cur.deleteCharAt(cur.length() - 1);
            }
        }
    }
}
```

```java
class Solution {
    List<String> res = new ArrayList<>();
    String[] keyboards = new String[]{
        "", "",
        "abc", "def",
        "ghi", "jkl", "mno",
        "pqrs", "tuv", "wxyz",
    };

    public List<String> letterCombinations(String digits) {
        if (digits == null || digits.length() == 0) {
            return res;
        }
        dfs(digits, 0, digits.length(), "");
        return res;
    }

    private void dfs(String digits, int pos, int len, String cur) {
        if (pos == len) {
            res.add(cur);
            return;
        } else {
            for (char c : keyboards[digits.charAt(pos) - '0'].toCharArray()) {
                dfs(digits, pos + 1, len, cur + c);
            }
        }
    }
}
```
