class Solution {
    boolean[] col, dg, udg;
    int n;

    public int totalNQueens(int n) {
        this.n = n;
        col = new boolean[n];
        dg = new boolean[n * 2];
        udg = new boolean[n * 2];

        return dfs(0);
    }

    private int dfs(int u) {
        if (u == n) {
            return 1;
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (!col[i] && !dg[u - i + n] && !udg[u + i]) {
                col[i] = dg[u - i + n] = udg[u + i] = true;
                res += dfs(u + 1);
                col[i] = dg[u - i + n] = udg[u + i] = false;
            }
        }
        return res;
    }
}
