class Solution {
    public int myAtoi(String s) {
        int k = 0;
        int n = s.length();
        while (k < n && s.charAt(k) == ' ') {
            k++;
        }
        if (k == n) {
            return 0;
        }

        int minus = 1;
        if (s.charAt(k) == '-') {
            minus = -1;
            k++;
        } else if (s.charAt(k) == '+') {
            k++;
        }

        int res = 0;
        while (k < n && s.charAt(k) >= '0' && s.charAt(k) <= '9') {
            int x = s.charAt(k) - '0';
            if (minus > 0 && res > (Integer.MAX_VALUE - x) / 10) {
                return Integer.MAX_VALUE;
            }
            if (minus < 0 && -res < (Integer.MIN_VALUE + x) / 10) {
                return Integer.MIN_VALUE;
            }
            if (-res * 10 - x == Integer.MIN_VALUE) {
                return Integer.MIN_VALUE;
            }
            res = res * 10 + x;
            k++;    
        }
        res *= minus;
        return res;
    }
}
