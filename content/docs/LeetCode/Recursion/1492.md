```java
    public int kthFactor(int n, int k) {
        return help(n, k, 1);
    }

    private int help(int n, int k, int f) {
        if (f >= n && k > 1) return -1;
        
        if (n % f == 0) {
            if (k == 1) return f;
            return help(n, k - 1, f + 1);
        }

        return help(n, k, f + 1);
    }
```
