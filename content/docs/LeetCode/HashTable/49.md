    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> map = new HashMap<>();
        for (String str : strs) {
            String oldStr = str;
            char[] chs = str.toCharArray();
            Arrays.sort(chs);
            str = new String(chs);

            List<String> tmp = map.getOrDefault(str, null);
            if (tmp == null) {
                tmp = new ArrayList<>();
                map.put(str, tmp);
            }
            tmp.add(oldStr);
        }

        List<List<String>> res = new ArrayList<>();
        for (var e : map.entrySet()) {
            res.add(e.getValue());
        }
        return res;
    }
