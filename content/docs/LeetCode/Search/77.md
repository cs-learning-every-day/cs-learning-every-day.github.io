class Solution {
    private List<List<Integer>> res;
    private int n;

    public List<List<Integer>> combine(int n, int k) {
        res = new ArrayList<>();
        this.n = n;
        dfs(1, k, new ArrayList<>());
        return res;
    }

    private void dfs(int s, int k, List<Integer> cur) {
        if (k == 0) {
            res.add(new ArrayList<>(cur));
            return;
        }

        for (int i = s; i <= n; i++) {
            cur.add(i);
            dfs(i + 1, k - 1, cur);
            cur.remove(cur.size() - 1);
        }
    }
}
