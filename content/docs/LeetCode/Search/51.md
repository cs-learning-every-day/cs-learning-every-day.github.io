class Solution {
public:
    vector<vector<string>> res;
    int n;
    vector<bool> col, dg, udg;
    vector<string> path;
    
    vector<vector<string>> solveNQueens(int _n) {
        n = _n;
        col = vector<bool>(n);
        dg = vector<bool>(n * 2);
        udg = vector<bool>(n * 2);
        path = vector<string>(n, string(n, '.'));

        dfs(0);
        return res;
    }

    void dfs(int u) {
        if (u == n) {
            res.push_back(path);
            return;
        }

        for (int i = 0; i < n; i++) {
            if (!col[i] && !dg[u + i] && !udg[u - i + n]) {
                col[i] = dg[u + i] = udg[u - i + n] = true;
                path[u][i] = 'Q';
                dfs(u + 1);
                path[u][i] = '.';
                col[i] = dg[u + i] = udg[u - i + n] = false;
            }
        }
    }
};
