class Solution {
public:
    int smallestDivisor(vector<int>& nums, int threshold) {
        int lo = 1, hi = 1e9;
        while (lo < hi) {
            int mid = (lo + hi) >> 1;
            if (check(nums, mid, threshold)) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }

    bool check(vector<int>& nums, int x, int threshold) {
        int res = 0;
        for (int n : nums) {
            if (n % x == 0) {
                res += (n / x);
            } else {
                res += (n / x) + 1;
            }
        }
        return res <= threshold;
    }
};
