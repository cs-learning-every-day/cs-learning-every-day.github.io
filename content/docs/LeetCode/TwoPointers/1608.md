    int specialArray(vector<int>& nums) {
        int lo = 0, hi = nums.size();
        while (lo <= hi) {
            int mid = (lo + hi) >> 1;
            int cnt = 0;
            for (int n : nums) {
                if (n >= mid) cnt++;
            }
            if (cnt == mid) {
                return mid;
            } else if (cnt > mid) {
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }
        return -1;
    }
