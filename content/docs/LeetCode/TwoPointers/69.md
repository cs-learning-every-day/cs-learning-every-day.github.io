    public int mySqrt(int x) {
        if (x == 0) return 0;

        int l = 1, r = x;

        while (l < r) {
            int mid = (int) ((l + 1L + r) >> 1);
            if (mid <= x / mid) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        
        return l;
    }
