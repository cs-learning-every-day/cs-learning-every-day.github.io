C++ map数组形式访问时会自动插入不存在的元数 值为默认
```c++
    string minWindow(string s, string t) {
        unordered_map<char, int> hs, ht;
        for (auto c : t) ht[c]++;
        
        string res = "";
        int cnt = 0;
        for (int i = 0, j = 0; i < s.size(); i++) {
            hs[s[i]]++;
            if (hs[s[i]] <= ht[s[i]]) {
                cnt++;
            }
            
            while (hs[s[j]] > ht[s[j]]) hs[s[j++]]--;

            if (cnt == t.size()) {
                if (!res.size() || i - j + 1 < res.size())
                    res = s.substr(j, i - j + 1);
            }
        }
        return res;
    }
```

```java
    public String minWindow(String s, String t) {
        Map<Character, Integer> hs = new HashMap<>();
        Map<Character, Integer> ht = new HashMap<>();

        for (var c : t.toCharArray()) ht.put(c, ht.getOrDefault(c, 0) + 1);

        int cnt = 0;
        String res = "";
        for (int i = 0, j = 0; i < s.length(); i++) {
            hs.put(s.charAt(i), hs.getOrDefault(s.charAt(i), 0) + 1);

            if (hs.get(s.charAt(i)) <= ht.getOrDefault(s.charAt(i), 0)) cnt++;

            while (j <= i && hs.get(s.charAt(j)) > ht.getOrDefault(s.charAt(j), 0)) {
                hs.put(s.charAt(j), hs.get(s.charAt(j)) - 1);
                j++;
            }
            if (cnt == t.length()) {
                if (res.length() == 0 || i - j + 1 < res.length())
                    res = s.substring(j, i + 1);
            }
        }
        return res;
    }
```
