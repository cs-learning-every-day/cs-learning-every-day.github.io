class Solution {
public:
    int minimizedMaximum(int n, vector<int>& quantities) {
        int lo = 1, hi = 1e5 + 10;
        while (lo < hi) {
            int mid = (lo + hi) >> 1;
            if (check(quantities, n, mid)) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }

    bool check(vector<int>& quantities, int n, int limit) {
        int count = 0;
        for (auto x : quantities) {
            if (x % limit == 0) {
                count += (x / limit);
            } else {
                count += (x / limit) + 1;
            }
        }
        return count <= n;
    }
};
