滑动窗口 + Hash

```java
    public List<Integer> findSubstring(String s, String[] words) {
        List<Integer> res = new ArrayList<>();
        int n = s.length(), m = words.length;
        if (m == 0) {
            return res;
        }
        int w = words[0].length();

        Map<String, Integer> tot = new HashMap<>();
        for (String word : words) tot.put(word, tot.getOrDefault(word, 0) + 1);

        for (int i = 0; i < w; i++) {
            Map<String, Integer> wd = new HashMap<>();
            int cnt = 0;
            for (int j = i; j + w <= n; j += w) {
                if (j >= i + m * w) {
                    String word = s.substring(j - m * w, j - (m - 1) * w);
                    wd.put(word, wd.get(word) - 1);
                    if (tot.containsKey(word) && 
                        wd.get(word) < tot.get(word)) cnt--;
                }

                String word = s.substring(j, j + w);
                wd.put(word, wd.getOrDefault(word, 0) + 1);

                if (tot.containsKey(word) && 
                    wd.get(word) <= tot.get(word)) cnt++;
                if (cnt == m) res.add(j - (m - 1) * w);
            }
        }

        return res;
    }
```
