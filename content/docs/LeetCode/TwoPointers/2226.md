假设x为每个小孩分到的糖果数，当x变大时堆数变小(即可满足的小孩数), x变小时堆变大。
```c++
class Solution {

typedef long long LL;

public:
    int maximumCandies(vector<int>& candies, long long k) {
        LL total = 0;
        for (auto c : candies) total += c;

        LL lo = 0, hi = total / k;
        while (lo < hi) {
            LL x = lo + hi + 1 >> 1;
            // check
            LL sum = 0;
            for (auto c : candies) {
                sum += (c / x);
            }
            if (sum >= k) {
                lo = x;
            } else {
                hi = x - 1;
            }
        }
        return lo;
    }
};
```
