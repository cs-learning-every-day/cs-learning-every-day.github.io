class Solution {
public:
    int minSpeedOnTime(vector<int>& dist, double hour) {
        int n = dist.size();
        // 速度超级快, 至少n-1个1hour
        if (hour <= n - 1) return -1;

        int lo = 1, hi = 1e7;
        while (lo < hi) {
            int x = (lo + hi) >> 1;
            if (check(dist, hour, x)) {
                hi = x;
            } else {
                lo = x + 1;
            }
        }
        return lo;
    }

    bool check(vector<int>& dist, double hour, int x) {
        double tmp = 0.0;
        for (int i = 0; i < dist.size() - 1; i++) {
            // time += (dist[i] - 1) / x + 1;
            tmp += std::ceil(dist[i] * 1.0 / x);
        }
        tmp += (dist.back() * 1.0 / x);
        return tmp <= hour;
    }
};
