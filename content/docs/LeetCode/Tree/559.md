```java
public int maxDepth(Node root) {
    if (root == null) return 0;
    if (root.children == null) return 1;
    
    int max = 0;
    int n = root.children.size();
    for (int i = 0; i < n; i++) {
        max = Math.max(max, maxDepth(root.children.get(i)));
    }
    return max + 1;
}
```
