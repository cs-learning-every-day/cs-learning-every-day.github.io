中序遍历
```c++
    vector<int> getAllElements(TreeNode* root1, TreeNode* root2) {
        vector<int> ans;
        vector<int> t1, t2;
        
        dfs(root1, t1);
        dfs(root2, t2);

        int i = 0, j = 0;
        for (; i < t1.size() && j < t2.size(); ) {
            if (t1[i] < t2[j]) {
                ans.push_back(t1[i++]);
            } else {
                ans.push_back(t2[j++]);
            }
        }
        while (i < t1.size()) {
            ans.push_back(t1[i++]);
        }
        while (j < t2.size()) {
            ans.push_back(t2[j++]);
        }
        return ans;
    }

    void dfs(TreeNode *root, vector<int> &t) {
        if (!root) return;
        dfs(root->left, t);
        t.push_back(root->val);
        dfs(root->right, t);
    }
```
