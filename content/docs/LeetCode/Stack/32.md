    public int longestValidParentheses(String s) {
        Stack<Integer> st = new Stack<>();
        int n = s.length();
        int result = 0;

        for (int i = 0, start = -1; i < n; i++) {
            if (s.charAt(i) == '(') {
                st.push(i);
            } else {
                if (st.isEmpty()) {
                    start = i;
                } else {
                    st.pop();
                    if (st.isEmpty()) {
                        result = Math.max(result, i - start);
                    } else {
                        result = Math.max(result, i - st.peek());
                    }
                }
            }
        }
        return result;
    }
