    public int findGCD(int[] nums) {
        Arrays.sort(nums);
        int x1 = nums[0];
        int x2 = nums[nums.length - 1];
        return gcd(x2, x1);
    }

    // a > b
    private int gcd(int a, int b) {
        if (a % b == 0) return b;
        return gcd(b, a % b);
    }
