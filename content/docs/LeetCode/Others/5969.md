class Solution {
    public boolean asteroidsDestroyed(int mass, int[] asteroids) {
        Arrays.sort(asteroids);
        long t = (long) mass;
        for (int a : asteroids) {
            if (t < a) {
                return false;
            } else {
                t += a;
            }
        }
        return true;
    }
}
