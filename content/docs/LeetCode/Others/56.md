class Solution {
    public int[][] merge(int[][] intervals) {
        int n = intervals.length;
        int[][] res = new int[n][2];
        
        Arrays.sort(intervals, (a, b) -> {
            if (a[0] == b[0]) {
                return a[1] - b[1];
            }
            return a[0] - b[0];
        });

        int idx = 0;
        int l = intervals[0][0], r = intervals[0][1];

        for (int i = 1; i < n; i++) {
            if (intervals[i][0] > r) {
                res[idx++] = new int[]{l, r};
                l = intervals[i][0];
                r = intervals[i][1];
            } else {
                r = Math.max(r, intervals[i][1]);
            }

        }
        res[idx++] = new int[]{l, r};
        return Arrays.copyOfRange(res, 0, idx);
    }
}
