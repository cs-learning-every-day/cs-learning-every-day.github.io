```java
    public int reverse(int x) {
        // java 这里-11 % 10 = -1 数学上应该是 9
        int flag = 1;
        if (x < 0) {
            flag = -1;
            x = -x;
        }

        int res = 0;
        while (x != 0) {
            if (res > ((Integer.MAX_VALUE - (x % 10)) / 10)) {
                return 0;
            }
            res = res * 10 + (x % 10);
            x /= 10;
        }
        return res * flag;
    }
```
