class Solution {
    public int[][] insert(int[][] a, int[] b) {
        if (a.length == 0) return new int[][]{{b[0], b[1]}};
        int n = a.length;
        int[][] res = new int[n + 1][2];
        int idx = 0;
        int k = 0;

        // 处理左边完全没有交集的情况
        while (k < n && a[k][1] < b[0]) {
            res[idx++] = new int[]{a[k][0], a[k][1]};
            k++;
        }

        // 处理交集
        if (k < n) {
            b[0] = Math.min(a[k][0], b[0]);
            while (k < n && a[k][0] <= b[1]) {
                b[1] = Math.max(b[1], a[k][1]);
                k++;
            }
        }
        res[idx++] = new int[]{b[0], b[1]};

        // 处理右边完全没有交集的情况
        while (k < n) {
            res[idx++] = new int[]{a[k][0], a[k][1]};
            k++;
        }

        return Arrays.copyOfRange(res, 0, idx);
    }
}
