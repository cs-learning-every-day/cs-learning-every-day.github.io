class Solution {
    private int res = 0;
    private int m;
    private int n;
    
    public int numberOfBeams(String[] bank) {
        m = bank.length;
        n = bank[0].length();
        for (int i = 0; i < m - 1; i++) {
            if (!bank[i].contains("1")) {
                continue;
            }
            int t = calculate(bank, i); // 计算这一行符合的激光束数量
            if (t > 0) { // update result
                for (int j = 0; j < n; j++) {
                    if (bank[i].charAt(j) == '1') {
                        res += t;
                    }
                }
            }
        }
        return res;
    }
    
    private int calculate(String[] bank, int start) {
        int end = start;
        for (int i = end + 1; i < m; i++) {
            if (bank[i].contains("1")) {
                end = i;
                break;
            }
        }
        int result = 0;
        if (end != start) {
            for (int i = 0; i < n; i++) {
                if (bank[end].charAt(i) == '1') {
                    result++;
                }
            }
        }
        
        return result;
    }
}
