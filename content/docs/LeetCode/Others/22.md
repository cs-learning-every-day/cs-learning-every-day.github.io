```java
class Solution {
    private List<String> res = new ArrayList<>();

    public List<String> generateParenthesis(int n) {
        dfs(n, 0, 0, "");
        return res;
    }

    private void dfs(int n, int open, int closed, String s) {
        if (open == n && closed == n) {
            res.add(s);
            return;
        }

        if (open < n) {
            dfs(n, open + 1, closed, s + "(");
        }

        if (closed < n && closed < open) {
            dfs(n, open, closed + 1, s + ")");
        }
    }
}
```
