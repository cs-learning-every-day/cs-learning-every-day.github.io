    public boolean checkString(String s) {
        char[] chs = s.toCharArray();
        int x = -1, y = -1;
        for (int i = 0; i < chs.length; i++) {
            if (chs[i] == 'a') {
                x = i;
            } else if (y == -1 && chs[i] == 'b') {
                y = i;
            }
        }
        
        if (x == -1 || y == -1) {
            return true;
        }
        return x < y;
    }
