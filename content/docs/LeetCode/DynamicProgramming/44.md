f[i][j]：表示s[0..i]和p[0..j]是否匹配

```java
    public boolean isMatch(String s, String p) {
        int m = s.length(), n = p.length();
        s = " " + s;
        p = " " + p;

        boolean[][] f = new boolean[m + 1][n + 1];
        f[0][0] = true;
        
        for (int i = 0; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (p.charAt(j) == '*') {
                    f[i][j] = (i > 0 && f[i - 1][j]) || f[i][j - 1];
                } else {
                    f[i][j] =  i > 0 && f[i - 1][j - 1] && 
                        (s.charAt(i) == p.charAt(j) || 
                         p.charAt(j) == '?');
                }
            }
        }

        return f[m][n];
    }
```
