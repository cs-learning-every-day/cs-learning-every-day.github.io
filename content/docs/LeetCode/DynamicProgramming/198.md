class Solution {
    public int rob(int[] nums) {
        int n = nums.length;
        // dp[0][0]: 第一个房间抢
        // dp[0][1]: 第一个房间不抢
        int[][] dp = new int[n][2];
        dp[0][0] = nums[0];

        for (int i = 1; i < n; i++) {
            dp[i][0] = dp[i - 1][1] + nums[i];
            dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][0]);
        }
        return Math.max(dp[n - 1][0], dp[n - 1][1]);
    }
}
