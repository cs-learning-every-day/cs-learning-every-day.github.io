class Solution {
    public int rob(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        }

        return Math.max(help(nums, 1, nums.length),
                        help(nums, 0, nums.length - 1));
    }
    // [i,j)
    private int help(int[] nums, int i, int j) {
        // 0 : 抢
        // 1 : 不抢
        int[][] dp = new int[nums.length][2];
        dp[i][0] = nums[i];

        while (++i < j) {
            dp[i][0] = dp[i - 1][1] + nums[i];
            dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][0]);
        }
        return Math.max(dp[j - 1][0], dp[j - 1][1]);
    }
}
