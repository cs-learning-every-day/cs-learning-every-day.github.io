    public int search(int[] nums, int target) {
        // 找到分隔点, 一半区间满足性质，另一半则不满足
        // 2 3 4 5 6 7, 1
        // 6 7, 1 2 3 4 5
        int l = 0, r = nums.length - 1;
        int mid;
        while (l < r) {
            mid = (l + r + 1) >> 1;
            if (nums[mid] >= nums[0]) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        // 判断在左半边、右半边
        if (target >= nums[0]) {
            l = 0;
        } else {
            l = r + 1;
            r = nums.length - 1;
        }

        while (l < r) {
            mid = (l + r) >> 1;
            if (nums[mid] >= target) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }

        if (nums[r] == target) return r;
        return -1;
    }
