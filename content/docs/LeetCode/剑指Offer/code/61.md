    bool isStraight(vector<int>& nums) {
        int min_ = 15, max_ = -1;
        unordered_set<int> set;
        for (auto n : nums) {
            if (n == 0) continue;
            min_ = min(n, min_);
            max_ = max(n, max_);
            if (set.find(n) != set.end()) return false;
            set.insert(n);
        }
        return max_ - min_ < 5;
    }
