    public boolean validateStackSequences(int[] pushed, int[] popped) {
        Stack<Integer> st = new Stack<>();
        int i = 0;
        for (var p : pushed) {
            st.push(p);
            while (!st.isEmpty() && st.peek() == popped[i]) {
                i++;
                st.pop();
            }
        }
        return st.isEmpty();
    }
