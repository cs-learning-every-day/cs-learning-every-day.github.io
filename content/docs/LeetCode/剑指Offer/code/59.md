    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        if (k == 1) return nums;
        vector<int> res;
        // 递减队列 
        deque<int> q;

        for (int i = 0; i < nums.size(); i++) {
            while (q.size() > 0 && nums[i] > q.back()) {
                q.pop_back();
            }
            q.push_back(nums[i]);
            if (i >= k - 1) { // 合法的窗口
                res.push_back(q.front());
                // 如果最大值是窗口第一个的就移除
                if (q.front() == nums[i - k + 1]) {
                    q.pop_front();
                }
            }
        }
        return res;
    }
