class CQueue {
private:
    stack<int> s1, s2;

public:
    CQueue() {

    }
    
    void appendTail(int value) {
        s1.push(value);
    }
    
    int deleteHead() {
        if (s1.empty()) return -1;
        while (s1.size() > 1) {
            s2.push(s1.top());
            s1.pop();
        }
        int res = s1.top();
        s1.pop();
        while (!s2.empty()) {
            s1.push(s2.top());
            s2.pop();
        }
        return res;
    }
};
