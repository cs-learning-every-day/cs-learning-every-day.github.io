```c++
    vector<int> exchange(vector<int>& nums) {
        for (int i = 0, j = nums.size() - 1; i <= j;) {
            if (nums[i] % 2 == 1) {
                i++;
            } else {
                swap(nums[i], nums[j]);
                j--;
            }
        }
        return nums;
    }
```

```java
    public int[] exchange(int[] nums) {
        int i = 0, j = nums.length - 1;
        while (i < j) {
            if (nums[i] % 2 == 0) {
                int tmp = nums[i];
                nums[i] = nums[j];
                nums[j] = tmp;
                j--;
            } else {
                i++;
            }
        }
        return nums;
    }
```
