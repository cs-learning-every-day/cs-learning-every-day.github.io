双指针
```c++
    int search(vector<int>& nums, int target) {
        if (nums.size() == 0) return 0;
        
        int l = 0, r = nums.size() - 1;
        while (l < r) {
            int m = (l + r) >> 1;
            if (nums[m] >= target) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        if (nums[l] != target) return 0;

        int i = l;

        l = 0;
        r = nums.size() - 1;
        while (l < r) {
            int m = (l + r + 1) >> 1;
            if (nums[m] <= target) {
                l = m;
            } else {
                r = m - 1;
            }
        }

        return l - i + 1;
    }
```
