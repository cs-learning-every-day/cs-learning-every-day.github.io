    ListNode* reverseList(ListNode* head) {
        auto dummy = new ListNode(-1);
        dummy->next = NULL;
        auto p = dummy;
        while (head != NULL) {
            auto t = head->next;
            head->next = p->next;
            p->next = head;
            head = t;
        }
        return dummy->next;
    }
