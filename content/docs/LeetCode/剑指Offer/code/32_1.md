class Solution {
    public int[] levelOrder(TreeNode root) {
        if (root == null) return new int[]{};
        
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);

        List<Integer> tmp = new ArrayList<>();

        while (!q.isEmpty()) {
            TreeNode node = q.poll();
            tmp.add(node.val);

            if (node.left != null) {
                q.offer(node.left);
            }

            if (node.right != null) {
                q.offer(node.right);
            }
        }

        int[] res = new int[tmp.size()];
        int idx = 0;
        for (int val : tmp) {
            res[idx++] = val;
        }
        return res;
    }
}
