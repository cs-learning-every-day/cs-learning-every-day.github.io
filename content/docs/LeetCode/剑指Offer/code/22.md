```c++
    ListNode* getKthFromEnd(ListNode* head, int k) {
        int n = 0;
        for (auto p = head; p != nullptr; p = p->next) {
            n++;
        }
        k %= n;
        if (k == 0) return head;
        auto p1 = head, p2 = head;
        while (k-- > 0) {
            p1 = p1->next;
        }
        while (p1) {
            p1 = p1->next;
            p2 = p2->next;
        }
        return p2;
    }
```

```java
    public ListNode getKthFromEnd(ListNode head, int k) {
        Stack<ListNode> s = new Stack<>();
        while (head != null) {
            s.push(head);
            head = head.next;
        }
        
        if (k > s.size()) return null;

        ListNode res = null;
        while (k > 0) {
            k--;
            res = s.pop();
        }
        return res;
    }
```
