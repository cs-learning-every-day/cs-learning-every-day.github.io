```c++
    vector<int> printNumbers(int n) {
        int max = 1;
        while (n -- > 0) max = max * 10;
        max--;
        vector<int> res(max, 0);
        for (int i = 1; i <= max; i++) res[i-1] = i;
        return res;
    }
```
