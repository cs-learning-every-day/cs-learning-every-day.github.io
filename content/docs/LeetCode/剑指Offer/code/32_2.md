```c++
    vector<int> levelOrder(TreeNode* root) {
        vector<int> res;
        queue<TreeNode*> q;
        if (root) q.push(root);

        while (!q.empty()) {
            auto n = q.size();
            for (int i = 0; i < n; i++) {
                auto p = q.front();
                q.pop();
                res.push_back(p->val);

                if (p->left) q.push(p->left);
                if (p->right) q.push(p->right);
            }
        }

        return res;
    }
```

```java
class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null) return res;

        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        q.offer(null);

        // [3, null, 9, 20, null, 15, 7, null]
        while (q.peek() != null) {
            TreeNode node;
            List<Integer> tmp = new ArrayList<>();
            while ((node = q.poll()) != null) {
                if (node.left != null) {
                    q.offer(node.left);
                }

                if (node.right != null) {
                    q.offer(node.right);
                }

                tmp.add(node.val);
            }
            res.add(tmp);
            q.offer(null);
        }

        return res;
    }
}
```
