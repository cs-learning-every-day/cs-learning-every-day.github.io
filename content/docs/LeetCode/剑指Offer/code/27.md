    TreeNode* mirrorTree(TreeNode* root) {
        if (!root) return root;
        TreeNode *leftRoot = mirrorTree(root->left);
        TreeNode *rightRoot = mirrorTree(root->right);

        root->left = rightRoot;
        root->right = leftRoot;
        return root;
    }
