class MinStack {
private:
    stack<int> s1, s2;

public:
    /** initialize your data structure here. */
    MinStack() {

    }
    
    void push(int x) {
        if (s2.empty() || x <= s2.top()) {
            s2.push(x);
        }
        s1.push(x);
    }
    
    void pop() {
        if (s1.top() == s2.top()) {
            s2.pop();
        }

        s1.pop();
    }
    
    int top() {
        return s1.top();
    }
    
    int min() {
        return s2.top();
    }
};
