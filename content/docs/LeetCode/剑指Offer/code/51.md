    int reversePairs(vector<int>& nums) {
        vector<int> tmp(nums.size(), 0);
        return mergeSort(nums, 0, nums.size() - 1, tmp);
    }
    
    int mergeSort(vector<int> &nums, int l, int r, vector<int> &tmp) {
        if (l >= r) return 0;
        int m = (l + r) >> 1;
        int result = mergeSort(nums, l, m, tmp) + mergeSort(nums, m + 1, r, tmp);

        int i = l, j = m + 1, k = 0;
        while (i <= m && j <= r) {
            if (nums[i] > nums[j]) {
                result += (m - i + 1);
                tmp[k++] = nums[j++];
            } else {
                tmp[k++] = nums[i++];
            }
        }
        while (i <= m) tmp[k++] = nums[i++];
        while (j <= r) tmp[k++] = nums[j++];
        for (i = l, k = 0; i <= r; i++, k++) nums[i] = tmp[k];
        return result;
    }
