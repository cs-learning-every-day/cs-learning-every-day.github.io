将元素放置到对应的位置。2代表下标为2的位置

```c++
    int findRepeatNumber(vector<int>& nums) {
        for (int i = 0; i < nums.size(); i++) {
            while (nums[i] != i) {
                if (nums[i] == nums[nums[i]]) return nums[i];
                swap(nums[i], nums[nums[i]]);
            }
        }
        return -1;
    }

    int findRepeatNumber(vector<int>& nums) {
        for (int i = 0; i < nums.size();) {
            if (nums[i] == i) {
                i++;
                continue;
            }

            if (nums[i] == nums[nums[i]]) return nums[i];

            swap(nums[i], nums[nums[i]]);
        }
        return -1;
    }
```
