将元素放置到对应的位置。2代表下标为2的位置

```c++
    int findRepeatNumber(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            while (i != nums[i] && nums[nums[i]] != nums[i]) {
                swap(nums[i], nums[nums[i]]);
            }
            if (i != nums[i] && nums[nums[i]] == nums[i]) {
                return nums[i];
            }
        }
        return -1;
    }
```
