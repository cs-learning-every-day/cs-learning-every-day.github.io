    bool isSymmetric(TreeNode* root) {
        if (!root) return true;
        return help(root->left, root->right);
    }

    bool help(TreeNode *A, TreeNode *B) {
        if (!A && !B) return true;
        if (!A || !B || A->val != B->val) return false;
        return help(A->left, B->right) && help(A->right, B->left);
    }
