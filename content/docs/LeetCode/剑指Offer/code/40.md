Quick Select k + 1
```c++
class Solution {
private:
    int oldK;
public:
    vector<int> getLeastNumbers(vector<int>& arr, int k) {
        if (k >= arr.size()) return arr;
        oldK = k;
        return quickSelect(arr, k, 0, arr.size() - 1);
    }

    vector<int> quickSelect(vector<int> &arr, int k, int l, int r) {
        if (k == 0) {
            vector<int> res;
            res.assign(arr.begin(), arr.begin() + oldK);
            return res;
        }
        int i = l - 1, j = r + 1, x = arr[l + r >> 1];
        while (i < j) {
            do i++; while (arr[i] < x);
            do j--; while (arr[j] > x);
            if (i < j) swap(arr[i], arr[j]);
        }
        int lCnt = j - l + 1;
        if (k < lCnt) return quickSelect(arr, k, l, j);
        else return quickSelect(arr, k - lCnt, j + 1, r);
    }
};
```
