```java
    public int lastRemaining(int n, int m) {
        int x = 0;
        for (int i = 2; i <= n; i++) {
            x = (x + m) % i;
        }
        return x;
    }
```

```java
    public int lastRemaining(int n, int m) {
        List<Integer> tmp = new LinkedList<>();
        for (int i = 0; i < n; i++) tmp.add(i);
        int idx = 0;
        while (n > 1) {
            idx = (idx + m - 1) % n;
            tmp.remove(idx);
            n--;
        }
        return tmp.get(0);
    }
```
