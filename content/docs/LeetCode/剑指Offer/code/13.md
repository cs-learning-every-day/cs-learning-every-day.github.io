简单DFS
```c++
class Solution {
private:
    int res_ = 0;
    int m_, n_, k_;
public:
    int movingCount(int m, int n, int k) {
        vector<vector<bool>> used(m, vector<bool>(n, false));
        m_ = m; n_ = n; k_ = k;
        dfs(0, 0, used);
        return res_;
    }
    
    int dx[4] = {-1, 1, 0, 0}, dy[4] = {0, 0, 1, -1};

    void dfs(int i, int j, vector<vector<bool>> &used) {
        if (i < 0 || j < 0 || i >= m_ || j >= n_ 
            || used[i][j]) return;
        if (calc(i, j) > k_) return;
        used[i][j] = true;
        res_++;
        for (int u = 0; u < 4; u++) {
            dfs(i + dx[u], j + dy[u], used);
        }
    }

    int calc(int x, int y) {
        int res = 0;
        while (x > 0) {
            res += x % 10;
            x /= 10;
        }
        while (y > 0) {
            res += y % 10;
            y /= 10;
        }
        return res;
    }
};
```

```java
class Solution {
    private int m;
    private int n;
    private int k;
    private boolean[][] visited;

    public int movingCount(int m, int n, int k) {
        this.m = m;
        this.k = k;
        this.n = n;
        this.visited = new boolean[m][n];

        return dfs(0, 0);
    }

    private int dfs(int x, int y) {
        if (x >= m || x < 0 || y >= n || y < 0) return 0;
        if (visited[x][y] || !isValid(x, y)) {
            return 0;
        }
        visited[x][y] = true;
        return 1 + dfs(x + 1, y) + 
            dfs(x, y + 1) + 
            dfs(x - 1, y) + 
            dfs(x, y - 1);
    }

    private boolean isValid(int x, int y) {
        int val = 0;
        while (x > 0) {
            val += (x % 10);
            x /= 10;
        }
        while (y > 0) {
            val += (y % 10);
            y /= 10;
        }
        return val <= k;
    }
}
```
