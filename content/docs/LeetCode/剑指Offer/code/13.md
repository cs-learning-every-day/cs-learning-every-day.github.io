简单DFS

```java
class Solution {
    private int m;
    private int n;
    private int k;
    private boolean[][] visited;

    public int movingCount(int m, int n, int k) {
        this.m = m;
        this.k = k;
        this.n = n;
        this.visited = new boolean[m][n];

        return dfs(0, 0);
    }

    private int dfs(int x, int y) {
        if (x >= m || x < 0 || y >= n || y < 0) return 0;
        if (visited[x][y] || !isValid(x, y)) {
            return 0;
        }
        visited[x][y] = true;
        return 1 + dfs(x + 1, y) + 
            dfs(x, y + 1) + 
            dfs(x - 1, y) + 
            dfs(x, y - 1);
    }

    private boolean isValid(int x, int y) {
        int val = 0;
        while (x > 0) {
            val += (x % 10);
            x /= 10;
        }
        while (y > 0) {
            val += (y % 10);
            y /= 10;
        }
        return val <= k;
    }
}
```
