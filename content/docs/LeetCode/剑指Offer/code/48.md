双指针

```java
public int lengthOfLongestSubstring(String s) {
    char[] chs = s.toCharArray();
    int res = 0;
    int i = 0, j = 0;

    Map<Character, Boolean> map = new HashMap<>();

    while (j <= i && i < chs.length) {
        if (map.getOrDefault(chs[i], false)) {
            res = Math.max(res, i - j);
            map.put(chs[j], false);
            j++;
        } else {
            map.put(chs[i], true);
            res = Math.max(res, i - j + 1);
            i++;
        }
    }
    return res;
}
```
