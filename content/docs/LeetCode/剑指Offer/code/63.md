    int maxProfit(vector<int>& prices) {
        if (prices.size() == 0) return 0;

        int res = 0;
        int minPrice = prices[0];
        for (auto p : prices) {
            res = max(res, p - minPrice);
            minPrice = min(minPrice, p);
        }
        return res;
    }
