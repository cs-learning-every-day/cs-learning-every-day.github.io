```c++
    typedef long long LL;
    int nthUglyNumber(int n) {
        if (n == 1) return 1;
        queue<LL> q1, q2, q3;
        q1.push(2);
        q2.push(3);
        q3.push(5);
        LL x = 0;
        while (--n > 0) {
            x = min(q1.front(), min(q2.front(), q3.front()));
            if (x == q1.front()) {
                q1.pop();
                q1.push(x * 2);
                q2.push(x * 3);
                q3.push(x * 5);
            } else if (x == q2.front()) {
                q2.pop();
                q2.push(x * 3);
                q3.push(x * 5);
            } else {
                q3.pop();
                q3.push(x * 5);
            }
        }
        return x;
    }
```
