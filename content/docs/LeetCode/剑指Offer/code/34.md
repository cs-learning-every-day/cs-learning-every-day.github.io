class Solution {
public:
    vector<vector<int>> res;
    vector<int> cur;
    vector<vector<int>> pathSum(TreeNode* root, int target) {
        dfs(root, target);
        return res;
    }

    void dfs(TreeNode *node, int target) {
        if (!node) return;
        int t = target - node->val;
        cur.push_back(node->val);

        if (t == 0 && !node->left && !node->right) {
            res.push_back(cur);
        }
        
        dfs(node->left, t);
        dfs(node->right, t);
        cur.pop_back();
    }
};
