    public String reverseWords(String s) {
        StringBuilder sb = new StringBuilder();
        char[] chs = s.toCharArray();
        int i = chs.length - 1, j = chs.length - 1;
        while (i >= 0) {
            while(i >= 0 && chs[i] != ' ') i--;
            sb.append(s.substring(i + 1, j + 1) + " ");
            while (i >= 0 && chs[i] == ' ') i--;
            j = i;
        }
        return sb.toString().trim();
    }
