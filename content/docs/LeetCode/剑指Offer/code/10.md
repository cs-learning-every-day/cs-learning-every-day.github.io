```c++
    int fib(int n) {
        if (n <= 1) return n;
        int a = 0, b = 1;
        int k = 1e9 + 7;
        for (int i = 2; i <= n; i++) {
            int t = b;
            b = (a + b) % k;
            a = t % k;
        }
        return b;
    }
```
