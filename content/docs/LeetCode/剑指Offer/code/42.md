    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        int res = nums[0];
        int prev = nums[0];
        for (int i = 1; i < n; i++) {
            prev = max(prev + nums[i], nums[i]);
            res = max(res, prev);
        }
        return res;
    }

    int maxSubArray(vector<int>& nums) {
        int n = nums.size();

        vector<int> dp(n, 0);
        dp[0] = nums[0];

        for (int i = 1; i < n; i++) {
            dp[i] = max(dp[i - 1] + nums[i], nums[i]);
        }
        int res = dp[0];
        for (int d : dp) res = max(d, res);
        return res;
    }
