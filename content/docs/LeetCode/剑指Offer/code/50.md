class Solution {
    public char firstUniqChar(String s) {
        Map<Character, Boolean> map = new HashMap<>();

        int[] counts = new int[26];
        char[] chs = s.toCharArray();
        for (char c : chs) {
            counts[c - 'a']++;
            map.put(c, !map.containsKey(c));
        }

        for (char c : chs) {
            if (map.get(c)) return c;
        }
        return ' ';
    }
}
