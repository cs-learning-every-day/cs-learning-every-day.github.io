滑动窗口

```c++
    vector<vector<int>> findContinuousSequence(int target) {
        vector<vector<int>> res;
        for (int i = 2, j = 1, s = 3; j < i;) {
            if (s >= target) {
                if (s == target) {
                    vector<int> tmp;
                    for (int k = j; k <= i; k++)
                        tmp.push_back(k);
                    res.push_back(tmp); 
                }
                s -= j;
                j++;
            } else if (s < target) {
                i++;
                s += i;
            }
        }
        return res;
    }
```
