    Node* copyRandomList(Node* head) {
        unordered_map<Node*, Node*> m;
        Node *dummy = new Node(-1);
        auto tail = dummy;
        for (auto p = head; p != nullptr; p = p->next) {
            tail->next = new Node(p->val);
            tail = tail->next;
            m[p] = tail;
        }
        for (auto p = head; p != nullptr; p = p->next) {
            m[p]->random = m[p->random];    
        }
        return dummy->next;
    }
