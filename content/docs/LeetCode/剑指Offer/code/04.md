class Solution {
    public boolean findNumberIn2DArray(int[][] matrix, int target) {
        if (matrix == null || matrix.length == 0) return false;
        int m = matrix.length;
        int n = matrix[0].length;

        int row = m - 1;
        int col = 0;
        while (row >= 0 && col < n) {
            if (matrix[row][col] == target) {
                return true;
            }

            if (matrix[row][col] < target) {
                col++;
            } else {
                row--;
            }
        }

        return false;
    }
}
