```c++
    bool searchArray(vector<vector<int>> array, int target) {
        if (array.size() == 0) {
            return false;
        }
        int m = array.size();
        int n = array[0].size();
        
        int i = 0, j = n - 1;
        while (i < m && j >= 0) {
            if (array[i][j] == target) {
                return true;
            } else if (array[i][j] < target) {
                i++;
            } else {
                j--;
            }
        }
        return false;
    }
```

```java
    public boolean findNumberIn2DArray(int[][] matrix, int target) {
        if (matrix == null || matrix.length == 0) return false;
        int m = matrix.length;
        int n = matrix[0].length;

        int row = m - 1;
        int col = 0;
        while (row >= 0 && col < n) {
            if (matrix[row][col] == target) {
                return true;
            }

            if (matrix[row][col] < target) {
                col++;
            } else {
                row--;
            }
        }

        return false;
    }
```
