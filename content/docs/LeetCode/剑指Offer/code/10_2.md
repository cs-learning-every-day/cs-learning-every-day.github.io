```c++
    int numWays(int n) {
        if (n <= 1) return 1;
        int k = 1000000007;
        int a = 1, b = 1;
        for (int i = 2; i <= n; i++) {
            int t = b;
            b = (a + b) % k;
            a = t;
        }
        return b;
    }
```
