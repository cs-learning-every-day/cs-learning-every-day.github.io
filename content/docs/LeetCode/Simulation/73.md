    public void setZeroes(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        // 将第一行和第一列看成 (m - 1) * (n - 1) 行和列为0的标识
        // 再用两个变量标识第一行和第一列
        int r1 = 1, c1 = 1;
        for (int i = 0; i < n; i++) {
            if (matrix[0][i] == 0) {
                r1 = 0;
                break;
            }
        }

        for (int i = 0; i < m; i++) {
            if (matrix[i][0] == 0) {
                c1 = 0;
                break;
            }
        }

        // 设置列
        for (int i = 1; i < m; i++)
            for (int j = 0; j < n; j++)
                if (matrix[i][j] == 0)
                    matrix[0][j] = 0;
        
        // 设置行
        for (int i = 0; i < n; i++)
            for (int j = 1; j < m; j++)
                if (matrix[j][i] == 0)
                    matrix[j][0] = 0;

        // 设置 (m - 1) * (n - 1)
        for (int i = 1; i < m; i++)
            if (matrix[i][0] == 0)
                for (int j = 1; j < n; j++)
                    matrix[i][j] = 0;  

        for (int i = 1; i < n; i++)
            if (matrix[0][i] == 0)
                for (int j = 1; j < m; j++)
                    matrix[j][i] = 0;

        if (r1 == 0) 
            for (int i = 0; i < n; i++)
                matrix[0][i] = 0;
    
        if (c1 == 0)
            for (int i = 0; i < m; i++)
                matrix[i][0] = 0;
    }
