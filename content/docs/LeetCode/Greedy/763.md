
先记录每个字符最后出现的位置，然后用两个指针表示区间并迭代，当到达最远的位置时表示当区间迭代完成
```c++
class Solution {
public:
    vector<int> partitionLabels(string s) {
        int n = s.size();
        vector<int> res, idxs(26, 0);
        for (int i = 0; i < n; i++) {
            idxs[s[i] - 'a'] = i;
        }
        int start = 0, last = 0;
        for (int i = 0; i < n; i++) {
            last = std::max(idxs[s[i] - 'a'], last);
            if (i == last) {
                res.push_back(last - start + 1);
                last = start = i + 1;
            }
        }
        return res;
    }
};
```
