快速选择
```java
class Solution {
    public int findKthLargest(int[] nums, int k) {
        return quickSelect(nums, k, 0, nums.length - 1);
    }
    
    private int quickSelect(int[] nums, int k, int l, int r) {
        if (l >= r) return nums[l];

        int i = l - 1, j = r + 1, x = nums[(l + r) >> 1];
        while (i < j) {
            do i++; while(nums[i] < x);
            do j--; while(nums[j] > x);
            if (i < j) {
                int tmp = nums[i];
                nums[i] = nums[j];
                nums[j] = tmp;
            }
        }
        // 右边区间元素个数
        if (r - j >= k) {
            return quickSelect(nums, k, j + 1, r);
        }
        return quickSelect(nums, k - (r - j), l, j);
    }
}
```

使用优先队列
```java
    // Time Complexity: O(nlgn)
    // Space Complexity: O(k)
    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> q = new PriorityQueue<>();
        for (int num : nums) {
            if (q.size() == k) {
                if (num > q.peek()) {
                    q.poll();
                    q.offer(num);
                }
            } else {
                q.offer(num);
            }
        }
        return q.peek();
    }
```
