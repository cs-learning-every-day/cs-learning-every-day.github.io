快速排序 Time Complexity: nlgn 最坏n^2  Space Complexity: O(n)
```java
class Solution {
    private ListNode[] arr;

    private void quickSort(int l, int r) {
        if (l >= r) return;
        int i = l - 1, j = r + 1;
        int x = arr[(l + r) >> 1].val;
        
        while (i < j) {
            do i++; while (arr[i].val < x);
            do j--; while (arr[j].val > x);
            if (i < j) {
                int t = arr[i].val;
                arr[i].val = arr[j].val;
                arr[j].val = t;
            }
        }
        
        quickSort(l, j);
        quickSort(j + 1, r);
    }

    public ListNode sortList(ListNode head) {
        int n = 0;
        for (ListNode t = head; t != null; t = t.next) n++;
        this.arr = new ListNode[n];
        int i = 0;
        for (ListNode t = head; t != null; t = t.next) arr[i++] = t;
        quickSort(0, n - 1);
        return head;
    }
}
```
