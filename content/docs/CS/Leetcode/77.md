    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        dfs(ans, 1, k, n, new ArrayList<>());
        return ans;
    }

    private void dfs(List<List<Integer>> ans, int start, int k, int n, List<Integer> cur) {
        if (cur.size() == k) {
            ans.add(new ArrayList<>(cur));
            return;
        }

        for (int i = start; i <= n; i++) {
            cur.add(i);
            dfs(ans, i + 1, k, n, cur);
            cur.remove(cur.size() - 1);
        }
    }
