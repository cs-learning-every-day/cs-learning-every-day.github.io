```c++
#include <iostream>
using namespace std;

const int N = 100010;

int idx, e[N], l[N], r[N];

void init() {
    // 0表示head 1表示tail
    r[0] = 1, l[1] = 0;
    idx = 2;
}

// 在下标为k的右边插入
void add(int k, int x) {
    e[idx] = x;
    l[idx] = k;
    r[idx] = r[k];
    l[r[k]] = idx;
    r[k] = idx;
    idx++;
}

void remove(int k) {
    r[l[k]] = r[k];
    l[r[k]] = l[k];
}


int main() {
    int n;
    cin >> n;
    
    init();
    
    while (n--) {
        string op;
        cin >> op;
        
        int k, a;
        
        if (op == "R") {
            cin >> a;
            add(l[1], a);
        } else if (op == "L") {
            cin >> a;
            add(0, a);
        } else if (op == "D") {
            cin >> k;
            remove(k + 1);
        } else if (op == "IL") {
            cin >> k >> a;
            add(l[k + 1], a);
        } else if (op == "IR") {
            cin >> k >> a;
            add(k + 1, a);
        }
    }
    for (int i = r[0]; i != 1; i = r[i]) cout << e[i] << " ";
    cout << endl;
    return 0;
}
```
