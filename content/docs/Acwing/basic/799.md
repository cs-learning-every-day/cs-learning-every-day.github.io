```c++
#include <iostream>
using namespace std;

int a[100010];
int s[100010];

int main() {
    int n;
    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    int maxLen = 0;
    for (int i = 0, j = 0; i < n; i++) {
        s[a[i]]++;
        while (s[a[i]] > 1) {
            s[a[j]]--;
            j++;
        }
        if ((i - j + 1) > maxLen) {
            maxLen = i - j + 1;
        }
    }
    cout << maxLen << endl;
    return 0;
}
```
