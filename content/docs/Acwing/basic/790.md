```c++
#include <iostream>

using namespace std;

int main() {
    double x;
    cin >> x;
    double l = -1e4, r = 1e4;
    while (r - l > 1e-8) {
        double m = (l + r) / 2;
        if (m * m * m >= x) {
            r =  m;
        } else {
            l = m;
        }
    }
    printf("%.6lf", r);
    return 0;
}
```
