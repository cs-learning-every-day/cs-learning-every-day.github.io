#include <iostream>
#include <vector>

using namespace std;

vector<int> add(vector<int> a, vector<int> b) {
    vector<int> v;
    
    int t = 0;
    for (int i = 0, j = 0; i < a.size() || j < b.size() || t;) {
        if (i < a.size()) t += a[i++];
        if (j < b.size()) t += b[j++];
        v.push_back(t % 10);
        t /= 10;
    }
    
    return v;
}

int main() {
    string n1, n2;
    cin >> n1 >> n2;
    vector<int> a, b;
    
    for (int i = n1.size() - 1; i >= 0; i--) a.push_back(n1[i] - '0');
    for (int i = n2.size() - 1; i >= 0; i--) b.push_back(n2[i] - '0');
    
    auto c = add(a, b);
    
    for (int i = c.size() - 1; i >= 0; i--) cout << c[i];
    cout << endl;
    
    return 0;
}
