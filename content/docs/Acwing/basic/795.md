```
S[i] = a[1] + a[2] + ... a[i]
a[l] + ... + a[r] = S[r] - S[l - 1]
```

```c++
#include <iostream>
using namespace std;

const int N = 100010;

int n, m;
int a[N], s[N];

int main() {
    a[0] = 0;
    s[0] = 0;

    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
    }

    for (int i = 1; i <= n; i++) {
        s[i] = s[i - 1] + a[i];
    }

    while (m--) {
        int l, r;
        scanf("%d%d", &r, &l);
        printf("%d\n", s[r] - s[l - 1]);
    }
    return 0;
}
```
