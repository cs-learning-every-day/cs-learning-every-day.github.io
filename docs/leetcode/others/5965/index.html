    class Solution {
    public long[] getDistances(int[] arr) {
        long[] res = new long[arr.length];
        
        for (int i = 0; i < arr.length; i++) {
            res[i] = calculate(arr, i);    
        }
        return res;
    }
    
    private int calculate(int[] arr, int begin) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < arr.length; i++) {
            if (i == begin) continue;
            if (arr[begin] == arr[i]) {
                list.add(i);
            }
        }
        
        int res = 0;
        for (int val : list) {
            res += Math.abs(begin - val);
        }
        return res;
    }
}
