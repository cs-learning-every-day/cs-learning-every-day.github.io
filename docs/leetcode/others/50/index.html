        public double myPow(double x, int n) {
        boolean isMinus = false;
        if (n < 0) {
            isMinus = true;
        }
        double res = 1;
        for (long k = (long)Math.abs((long)n); k != 0; k >>= 1) {
            if ((k & 0x1) == 0x1) {
                res *= x;
            }
            x *= x;
        }
        if (isMinus) {
            return 1 / res;
        } else {
            return res;
        }
    }
