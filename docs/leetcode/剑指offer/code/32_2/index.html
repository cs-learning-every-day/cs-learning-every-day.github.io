    class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        if (root == null) return res;

        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        q.offer(null);

        // [3, null, 9, 20, null, 15, 7, null]
        while (q.peek() != null) {
            TreeNode node;
            List<Integer> tmp = new ArrayList<>();
            while ((node = q.poll()) != null) {
                if (node.left != null) {
                    q.offer(node.left);
                }

                if (node.right != null) {
                    q.offer(node.right);
                }

                tmp.add(node.val);
            }
            res.add(tmp);
            q.offer(null);
        }

        return res;
    }
}
