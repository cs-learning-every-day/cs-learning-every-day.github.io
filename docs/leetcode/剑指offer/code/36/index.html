    class Solution {
public:
    Node* treeToDoublyList(Node* root) {
        if (!root) return nullptr;
        prev = nullptr; head = nullptr;
        dfs(root);
        head->left = prev;
        prev->right = head;
        return head;
    }
private:
    Node *prev, *head;
    void dfs(Node *node) {
        if (!node) return;
        dfs(node->left);
        if (head == nullptr) {
            head = prev = node;
        } else {
            prev->right = node;
            node->left = prev;
            prev = node;
        }
        dfs(node->right);
    }
};
