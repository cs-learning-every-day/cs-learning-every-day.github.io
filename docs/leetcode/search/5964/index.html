    class Solution {
    private int n;
    
    public int[] executeInstructions(int n, int[] startPos, String s) {
        int tmp = s.length();
        int[] res = new int[tmp];
        int idx = 0;
        this.n = n;
        
        
        for (int i = 0; i < tmp; i++) {
            res[idx++] = help(Arrays.copyOf(startPos, 2), s, i, tmp);
        }
        return res;
    }
    
    private int help(int[] pos, String s, int start, int len) {
        int res = 0;
        for (int i = start; i < len; i++) {
            if (checkAndRun(pos, s.charAt(i))) {
                res++;
            } else {
                break;
            }
        }
        return res;
    }
    
    private boolean checkAndRun(int[] pos, char c) {
        if (c == 'U') {
            if (pos[0] != 0) {
                pos[0]--;
                return true;
            }
        } else if (c == 'R') {
            if (pos[1] != (n - 1)) {
                pos[1]++;
                return true;
            }
        } else if (c == 'L') {
            if (pos[1] != 0) {
                pos[1]--;
                return true;
            }
        } else if (c == 'D') {
            if (pos[0] != (n - 1)) {
                pos[0]++;
                return true;
            }
        }
        return false;
    }
}
