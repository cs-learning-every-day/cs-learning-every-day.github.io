    DP

    dp[i][j] 表示能到i行j列不同路径个数

    初始化第一行和和第一列为1

    class Solution {
    public int uniquePaths(int m, int n) {
        int[][] dp = new int[m + 1][n + 1];
        for (int i = 1; i <= m; i++) {
            dp[i][1] = 1;
        }
        for (int j = 1; j <= n; j++) {
            dp[1][j] = 1;
        }

        for (int i = 2; i <= m; i++) {
            for (int j = 2; j <= n; j++) {
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
        
        return dp[m][n];
    }
}

    DFS+记忆化

    class Solution {
    public int uniquePaths(int m, int n) {
        int[][] memo =  new int[m][n];

        return dfs(0, 0, m, n, memo);
    }

    private int dfs(int i, int j, int m, int n, int[][] memo) {
        if (i >= m || j >= n || i < 0 || j < 0) {
            return 0;
        }

        if (memo[i][j] != 0) {
            return memo[i][j];
        }

        if (i == m - 1 && j == n - 1) {
            return 1;
        }

        int res = dfs(i + 1, j, m, n, memo) + dfs(i, j + 1, m, n, memo);
        memo[i][j] = res;
        return res;
    }
}
