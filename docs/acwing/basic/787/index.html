    #include <iostream>

using namespace std;
const int N = 100010;
int q[N], tmp[N];

void mergeSort(int l, int r) {
    if (l >= r) return;
    
    int mid = l + r >> 1;
    mergeSort(l, mid), mergeSort(mid + 1, r);
    
    int i = l, j = mid + 1, k = 0;
    while (i <= mid && j <= r) {
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];
    
    for (int i = l, j = 0; i <= r; i++, j++) q[i] = tmp[j];
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) cin >> q[i];
    mergeSort(0, n - 1);
    for (int i = 0; i < n; i++) cout << q[i] << " ";
    return 0;
}
