vecotr<int> mul(vecotr<int> &A, int b) {
    vecotr<int> C;
    int t = 0;
    for (int i = 0; i < A.size() || t != 0; i++) {
        if (i < A.size()) {
            t += A[i] * b;
        }
        C.push_back(t % 10);
        t /= 10;
    }

    return C;
}
