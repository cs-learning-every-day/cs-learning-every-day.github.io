    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> ans;
        vector<int> cur;
        vector<bool> used(nums.size(), false);
        dfs(ans, nums, 0, cur, used);
        return ans;
    }
private:
    void dfs(vector<vector<int>>& ans, vector<int>& nums, int start, vector<int>& cur, vector<bool>& used) {
        if (cur.size() == nums.size()) {
            ans.push_back(cur);
            return;
        }

        for (int i = 0; i < nums.size(); i++) {
            if (used[i]) continue;
            used[i] = true;
            cur.push_back(nums[i]);
            dfs(ans, nums, i + 1, cur, used);
            cur.pop_back();
            used[i] = false;
        }
    }
